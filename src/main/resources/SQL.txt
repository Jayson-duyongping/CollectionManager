创建用户表
CREATE TABLE IF NOT EXISTS t_user(
id INT UNSIGNED AUTO_INCREMENT,
username VARCHAR(40) NOT NULL,
password VARCHAR(100) NOT NULL,
headicon VARCHAR(100),
phonenumber VARCHAR(20),
sina VARCHAR(20),
qq VARCHAR(20),
wx VARCHAR(50),
nickname VARCHAR(50),
sex VARCHAR(10) DEFAULT '1',
birthday VARCHAR(20),
sketch VARCHAR(50),
address VARCHAR(100),
description VARCHAR(200),
hobby VARCHAR(200),
skill VARCHAR(200),
PRIMARY KEY(id)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;


创建动态表(关联外键t_user)
CREATE TABLE IF NOT EXISTS t_dynamic(
id INT UNSIGNED AUTO_INCREMENT,
user_id INT UNSIGNED,
type VARCHAR(20) NOT NULL,
title VARCHAR(40),
content VARCHAR(200),
imgs VARCHAR(200),
musics VARCHAR(200),
time VARCHAR(50),
FOREIGN KEY(user_id) REFERENCES t_user(id),
PRIMARY KEY(id)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
